# InfluxDB Backup Cron Jobs
# This file is processed by the entrypoint script to configure cron jobs

# Daily incremental backup at 2:00 AM
0 2 * * * backup /usr/local/bin/backup.sh incremental >> /var/log/backup.log 2>&1

# Weekly full backup on Sundays at 3:00 AM
0 3 * * 0 backup /usr/local/bin/backup.sh full >> /var/log/backup.log 2>&1

# Daily backup cleanup at 4:00 AM
0 4 * * * backup /usr/local/bin/backup_utils.sh cleanup >> /var/log/backup.log 2>&1

# Weekly backup health check on Saturdays at 5:00 AM
0 5 * * 6 backup /usr/local/bin/backup_utils.sh stats >> /var/log/backup.log 2>&1
